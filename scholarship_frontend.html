// Student Scholarship Disbursement — Soroban
// ═══════════════════════════════════════════════
#![no_std]

use soroban_sdk::{
    contract, contractimpl, contracttype,
    Address, Env,
};

#[contracttype]
pub enum DataKey {
    Admin,
    PoolLocked,
    PoolDisbursed,
    PerSemesterAmt,
    MinGpaScaled,     // ×100 e.g. 250 = 2.50
    MaxSemesters,
    Student(u64),
    Disbursed(u64, u32),
}

#[contracttype]
pub struct Student {
    pub id:               u64,
    pub gpa_scaled:       u32,
    pub current_semester: u32,
    pub total_semesters:  u32,
    pub is_active:        bool,
    pub total_received:   u64,
}

#[contract]
pub struct ScholarshipContract;

#[contractimpl]
impl ScholarshipContract {

  pub fn release_funds(
      env: Env,
      caller: Address,
      student_id: u64,
  ) -> u64 {
      caller.require_auth();
      Self::assert_admin(&env, &caller);

      let mut student: Student =
          env.storage().persistent()
             .get(&DataKey::Student(student_id))
             .expect("student not found");

      // ── Gate 1: active ──
      if !student.is_active {
          panic!("student is not active");
      }

      // ── Gate 2: semester limit ──
      let next_sem = student.current_semester + 1;
      let max: u32 = env.storage().instance()
          .get(&DataKey::MaxSemesters).unwrap();
      if next_sem > max { panic!("semester limit exceeded"); }

      // ── Gate 3: performance ──
      let min_gpa: u32 = env.storage().instance()
          .get(&DataKey::MinGpaScaled).unwrap();
      if student.current_semester > 0
          && student.gpa_scaled < min_gpa
      {
          panic!("gpa below threshold");
      }

      // ── Gate 4: duplicate guard ──
      let dis_key = DataKey::Disbursed(student_id, next_sem);
      if env.storage().persistent().has(&dis_key) {
          panic!("already disbursed");
      }

      // ── Gate 5: solvency ──
      let locked: u64 = env.storage().instance()
          .get(&DataKey::PoolLocked).unwrap();
      let disbursed: u64 = env.storage().instance()
          .get(&DataKey::PoolDisbursed).unwrap();
      let amount: u64 = env.storage().instance()
          .get(&DataKey::PerSemesterAmt).unwrap();

      if locked < disbursed + amount {
          panic!("insufficient funds");
      }

      // ── Release ──
      env.storage().instance()
         .set(&DataKey::PoolDisbursed, &(disbursed + amount));
      env.storage().persistent()
         .set(&dis_key, &true);

      student.current_semester = next_sem;
      student.total_received  += amount;
      env.storage().persistent()
         .set(&DataKey::Student(student_id), &student);

      amount
  }
}

// ── Storage Layout ──────────────────────────
//
// Instance Storage (contract-level config)
// ─────────────────────────────────────────
DataKey::Admin          →  Address
DataKey::PoolLocked     →  u64     // total funds deposited
DataKey::PoolDisbursed  →  u64     // running total released
DataKey::PerSemesterAmt →  u64     // base release per semester
DataKey::MinGpaScaled   →  u32     // e.g. 250 = 2.50
DataKey::MaxSemesters   →  u32     // global cap

//
// Persistent Storage (per-entity, survives ledger TTL)
// ────────────────────────────────────────────────────
DataKey::Student(id)         →  Student {
    id: u64,
    gpa_scaled: u32,
    current_semester: u32,
    total_semesters: u32,
    is_active: bool,
    total_received: u64,
}
DataKey::Disbursed(id, sem)  →  bool  // idempotency guard

//
// GPA Scaling Convention
// ─────────────────────
// All GPAs stored as integers × 100
// 0.00 GPA  →  0
// 2.50 GPA  →  250   (minimum threshold)
// 3.75 GPA  →  375
// 4.00 GPA  →  400   (maximum)
